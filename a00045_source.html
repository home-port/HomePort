76 static void on_ev_recv(hpd_ev_loop_t *loop, struct ev_io *watcher, int revents)
77 {
78  ssize_t received;
79  hpd_tcpd_conn_t *conn = watcher->data;
80  hpd_tcpd_settings_t *settings = &conn->tcpd->settings;
81  size_t max_data_size = settings->max_data_size;
82  char buffer[max_data_size];
83  hpd_module_t *context = conn->tcpd->context;
84 
85  HPD_LOG_VERBOSE(context, "Receiving data from %s", conn->ip);
86  if ((received = recv(watcher->fd, buffer, max_data_size-1, 0)) < 0) {
87  int err = errno;
88  if (received == -1 && errno == EWOULDBLOCK) {
89  HPD_LOG_WARN(context, "libev callback called without data to receive (conn: %s)", conn->ip);
90  return;
91  }
92  HPD_LOG_ERROR(context, "recv(): %s", strerror(err));
93  if (hpd_tcpd_conn_kill(conn)) HPD_LOG_ERROR(context, "Failed to kill connection.");
94  return;
95  } else if (received == 0) {
96  HPD_LOG_INFO(context, "Connection closed by %s\n", conn->ip);
97  if (hpd_tcpd_conn_kill(conn)) HPD_LOG_ERROR(context, "Failed to kill connection.");
98  return;
99  }
100 
101  if (settings->on_receive) {
102  if (settings->on_receive(conn->tcpd, conn,
103  settings->tcpd_ctx, &conn->ctx,
104  buffer, (size_t) received)) {
105  HPD_LOG_ERROR(context, "Failed to handle new data, killing it.");
106  if (hpd_tcpd_conn_kill(conn)) HPD_LOG_ERROR(context, "Failed to kill connection.");
107  return;
108  }
109  }
110 
111  // Reset timeout
112  if (conn->timeout) {
113  conn->timeout_watcher.repeat = conn->tcpd->settings.timeout;
114  ev_timer_again(loop, &conn->timeout_watcher);
115  }
116 }

130 static void on_ev_send(hpd_ev_loop_t *loop, struct ev_io *watcher, int revents)
131 {
132  hpd_tcpd_conn_t *conn = watcher->data;
133  ssize_t sent;
134  hpd_module_t *context = conn->tcpd->context;
135 
136  sent = send(watcher->fd, conn->send_msg, conn->send_len, 0);
137  if (sent < 0) {
138  int err = errno;
139  HPD_LOG_ERROR(context, "send(): %s", strerror(err));
140  } else if (sent == conn->send_len) {
141  free(conn->send_msg);
142  conn->send_msg = NULL;
143  conn->send_len = 0;
144  } else {
145  conn->send_len -= sent;
146  char *s = malloc(conn->send_len*sizeof(char));
147  if (!s) {
148  HPD_LOG_ERROR(context, "Cannot allocate enough memory.");
149  free(conn->send_msg);
150  conn->send_msg = NULL;
151  conn->send_len = 0;
152  } else {
153  strcpy(s, &conn->send_msg[sent]);
154  free(conn->send_msg);
155  conn->send_msg = s;
156  return;
157  }
158  }
159 
160  ev_io_stop(conn->tcpd->loop, &conn->send_watcher);
161  if (conn->send_close && hpd_tcpd_conn_kill(conn)) HPD_LOG_ERROR(context, "Failed to kill connection.");
162 }

195 static void on_ev_conn(hpd_ev_loop_t *loop, struct ev_io *watcher, int revents)
196 {
197  char ip_string[INET6_ADDRSTRLEN];
198  int in_fd;
199  socklen_t in_size;
200  struct sockaddr_storage in_addr_storage;
201  struct sockaddr *in_addr = (struct sockaddr *)&in_addr_storage;
202  hpd_tcpd_conn_t *conn;
203  hpd_tcpd_t *tcpd = (hpd_tcpd_t *)watcher->data;
204  hpd_tcpd_settings_t *settings = &tcpd->settings;
205  hpd_module_t *context = tcpd->context;
206 
207  // Accept connection
208  in_size = sizeof *in_addr;
209  if ((in_fd = accept(watcher->fd, in_addr, &in_size)) < 0) {
210  int err = errno;
211  HPD_LOG_ERROR(context, "accept(): %s", strerror(err));
212  return;
213  }
214 
215  // Print a nice message
216  inet_ntop(in_addr_storage.ss_family, get_in_addr(in_addr), ip_string, sizeof ip_string);
217  HPD_LOG_INFO(context, "Got connection from %s.", ip_string);
218 
219  // Create conn and parser
220  conn = malloc(sizeof(hpd_tcpd_conn_t));
221  if (conn == NULL) {
222  HPD_LOG_ERROR(context, "Cannot allocation memory for connection.");
223  close(in_fd);
224  return;
225  }
226  conn->tcpd = watcher->data;
227  strcpy(conn->ip, ip_string);
228  conn->timeout_watcher.data = conn;
229  conn->recv_watcher.data = conn;
230  conn->send_watcher.data = conn;
231  conn->ctx = NULL;
232  conn->send_msg = NULL;
233  conn->send_len = 0;
234  conn->send_close = 0;
235  conn->timeout = 1;
236 
237  // Add connection to list
238  TAILQ_INSERT_TAIL(&conn->tcpd->conns, conn, HPD_TAILQ_FIELD);
239 
240  // Call back
241  if (settings->on_connect) {
242  if (settings->on_connect(conn->tcpd, conn, settings->tcpd_ctx, &conn->ctx)) {
243  HPD_LOG_ERROR(context, "Failed to handle new connection, killing it.");
244  if (hpd_tcpd_conn_kill(conn)) HPD_LOG_ERROR(context, "Failed to kill connection.");
245  return;
246  }
247  }
248 
249  // Start timeout and io watcher
250  ev_io_init(&conn->recv_watcher, on_ev_recv, in_fd, EV_READ);
251  ev_io_init(&conn->send_watcher, on_ev_send, in_fd, EV_WRITE);
252  ev_io_start(loop, &conn->recv_watcher);
253  ev_init(&conn->timeout_watcher, on_ev_timeout);
254  conn->timeout_watcher.repeat = settings->timeout;
255  if (conn->timeout)
256  ev_timer_again(loop, &conn->timeout_watcher);
257 }

320 hpd_error_t hpd_tcpd_start(hpd_tcpd_t *tcpd)
321 {
322  if (!tcpd) return HPD_E_NULL;
323 
324  int rc;
325  struct addrinfo hints;
326  struct addrinfo *servinfo, *p;
327 
328  // Clear struct and set requirements for socket
329  memset(&hints, 0, sizeof hints);
330  hints.ai_family = AF_UNSPEC; // IPv4/IPv6
331  hints.ai_socktype = SOCK_STREAM; // TCP Stream
332  hints.ai_flags = AI_PASSIVE; // Wildcard address
333 
334  // Get address infos we later use this to open socket
335  if ((rc = getaddrinfo(NULL, tcpd->port_str, &hints, &servinfo)) != 0) {
336  switch (rc) {
337  case EAI_SYSTEM: {
338  int err = errno;
339  HPD_LOG_ERROR(tcpd->context, "getaddrinfo() failed with: %s", gai_strerror(rc));
340  HPD_LOG_ERROR(tcpd->context, "System error message: %s", strerror(err));
341  freeaddrinfo(servinfo);
342  return HPD_E_UNKNOWN;
343  }
344  default: {
345  HPD_LOG_ERROR(tcpd->context, "getaddrinfo() failed with: %s", gai_strerror(rc));
346  freeaddrinfo(servinfo);
347  return HPD_E_UNKNOWN;
348  }
349  }
350  }
351 
352  // Loop through results and bind to first
353  for (p = servinfo; p != NULL; p=p->ai_next) {
354 
355  // Create socket
356  if ((tcpd->sockfd = socket(p->ai_family, p->ai_socktype, p->ai_protocol)) < 0) {
357  int err = errno;
358  HPD_LOG_DEBUG(tcpd->context, "socket() failed with: %s", strerror(err));
359  continue;
360  }
361 
362  // Change to non-blocking sockets
363  if (fcntl(tcpd->sockfd, F_SETFL, O_NONBLOCK)) {
364  int err = errno;
365  HPD_LOG_DEBUG(tcpd->context, "fcntl() failed with: %s", strerror(err));
366  close(tcpd->sockfd);
367  continue;
368  }
369 
370  // Reuse addr for testing purposes
371 #ifdef DEBUG
372  int yes = 1;
373  if (setsockopt(tcpd->sockfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) == -1) {
374  int err = errno;
375  HPD_LOG_DEBUG(tcpd->context, "setsockopt() failed with: %s", strerror(err));
376  close(tcpd->sockfd);
377  continue;
378  }
379 #endif
380 
381  // Bind to socket
382  if (bind(tcpd->sockfd, p->ai_addr, p->ai_addrlen) != 0) {
383  int err = errno;
384  HPD_LOG_DEBUG(tcpd->context, "bind() failed with: %s", strerror(err));
385  close(tcpd->sockfd);
386  continue;
387  }
388 
389  break;
390  }
391 
392  // Check if we binded to anything
393  if (p == NULL) {
394  close(tcpd->sockfd);
395  freeaddrinfo(servinfo);
396  return HPD_E_UNKNOWN;
397  }
398 
399  // Clean up
400  freeaddrinfo(servinfo);
401 
402  // Listen on socket
403  if (listen(tcpd->sockfd, SOMAXCONN) < 0) {
404  int err = errno;
405  HPD_LOG_DEBUG(tcpd->context, "listen() failed with: %s", strerror(err));
406  close(tcpd->sockfd);
407  return HPD_E_UNKNOWN;
408  }
409 
410  // Set listener on libev
411  tcpd->watcher.data = tcpd;
412  ev_io_init(&tcpd->watcher, on_ev_conn, tcpd->sockfd, EV_READ);
413  ev_io_start(tcpd->loop, &tcpd->watcher);
414 
415  return HPD_E_SUCCESS;
416 }

430 hpd_error_t hpd_tcpd_stop(hpd_tcpd_t *tcpd)
431 {

433 
434  hpd_tcpd_conn_t *conn, *tmp;

449  }
450 
451  return HPD_E_SUCCESS;

536 hpd_error_t hpd_tcpd_conn_vsendf(hpd_tcpd_conn_t *conn, const char *fmt, va_list vp)
537 {
538  if (!conn) return HPD_E_NULL;
539 
540  char *new_msg;
541  int new_len;
542  va_list vp2;
543 
544  // Copy arg to avoid errors on 64bit
545  va_copy(vp2, vp);
546 
547  // Get the length to expand with
548  new_len = vsnprintf("", 0, fmt, vp);
549  if (new_len < 0) {
550  int err = errno;
551  HPD_LOG_DEBUG(conn->tcpd->context, "vsnprintf(): %s", strerror(err));
552  return HPD_E_UNKNOWN;
553  }
554 
555  // Expand message to send
556  new_msg = realloc(conn->send_msg,
557  (conn->send_len + new_len + 1)*sizeof(char));
558  if (new_msg == NULL) HPD_LOG_RETURN_E_ALLOC(conn->tcpd->context);
559  conn->send_msg = new_msg;
560 
561  // Concatenate strings
562  vsprintf(&(conn->send_msg[conn->send_len]), fmt, vp2);
563 
564  // Start send watcher
565  if (conn->send_len == 0 && conn->tcpd != NULL)
566  ev_io_start(conn->tcpd->loop, &conn->send_watcher);
567 
568  // Update length
569  conn->send_len += new_len;
570 
571  return HPD_E_SUCCESS;
572 }

610 hpd_error_t hpd_tcpd_conn_kill(hpd_tcpd_conn_t *conn)
611 {
612  if (!conn) return HPD_E_NULL;
613 
614  hpd_tcpd_settings_t *settings = &conn->tcpd->settings;
615 
616  // Print messange
617  HPD_LOG_INFO(conn->tcpd->context, "Killing connection from %s.", conn->ip);
618 
619  // Stop circular calls and only kill this connection once
620  if (conn->recv_watcher.fd < 0) return HPD_E_SUCCESS;
621 
622  // Stop watchers
623  ev_io_stop(conn->tcpd->loop, &conn->recv_watcher);
624  ev_io_stop(conn->tcpd->loop, &conn->send_watcher);
625  ev_timer_stop(conn->tcpd->loop, &conn->timeout_watcher);
626 
627  // Close socket
628  if (close(conn->recv_watcher.fd) != 0) {
629  int err = errno;
630  HPD_LOG_ERROR(conn->tcpd->context, "close(): %s", strerror(err));
631  }
632  conn->recv_watcher.fd = -1;
633 
634  // Remove from list
635  TAILQ_REMOVE(&conn->tcpd->conns, conn, HPD_TAILQ_FIELD);
636 
637  // Call back
638  if (settings->on_disconnect && settings->on_disconnect(conn->tcpd, conn, settings->tcpd_ctx, &conn->ctx))
639  HPD_LOG_ERROR(conn->tcpd->context, "Failed to disconnect.");
640 
641  // Cleanup
642  free(conn->send_msg);
643  free(conn);
644 
645  return HPD_E_SUCCESS;
646 }

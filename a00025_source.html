82 static content_type_t media_type_to_enum(const char *haystack)
83 {
84  char *str;
85 
86  if (haystack == NULL)
87  return CONTENT_NONE;
88  if ( strcmp(haystack, "application/xml") == 0 ||
89  strncmp(haystack, "application/xml;", 16) == 0 ||
90  ((str = strstr(haystack, ",application/xml")) && (str[16] == ';' || str[16] == '\0')))
91  return CONTENT_XML;
92  if ( strcmp(haystack, "application/json") == 0 ||
93  strncmp(haystack, "application/json;", 17) == 0 ||
94  ((str = strstr(haystack, ",application/json")) && (str[17] == ';' || str[17] == '\0')))
95  return CONTENT_JSON;
96  if (strcmp(haystack, "*/*") == 0)
97  return CONTENT_WILDCARD;
98  return CONTENT_UNKNOWN;
99 }

128 
129  hpd_adapter_id_t *adapter = NULL;
130  hpd_device_id_t *device = NULL;
131  if ((rc = hpd_service_get_adapter(service, &adapter))) goto id_error;
132  if ((rc = hpd_service_get_device(service, &device))) goto id_error;
133 
134  const char *adp_id, *dev_id, *srv_id;
135  if ((rc = hpd_service_get_id(service, &srv_id))) goto id_error;
136  if ((rc = hpd_device_get_id(device, &dev_id))) goto id_error;
137  if ((rc = hpd_adapter_get_id(adapter, &adp_id))) goto id_error;
138 
139  if ((rc = hpd_device_id_free(device))) {
140  device = NULL;
141  goto id_error;
142  }
143  device = NULL;
144  if ((rc = hpd_adapter_id_free(adapter))) {
145  adapter = NULL;
146  goto id_error;
147  }
148  adapter = NULL;
149 
150  char *aid = NULL, *did = NULL, *sid = NULL;
151  if ((rc = url_encode(rest, adp_id, &aid))) goto encode_error;
152  if ((rc = url_encode(rest, dev_id, &did))) goto encode_error;
153  if ((rc = url_encode(rest, srv_id, &sid))) goto encode_error;
154 
155  (*url) = malloc((strlen(aid)+strlen(did)+strlen(sid)+3+1)*sizeof(char));
156  if (!(*url)) {
157  rc = HPD_E_ALLOC;
158  goto encode_error;
159  }
160  (*url)[0] = '\0';
161 
162  strcat((*url), "/");
163  strcat((*url), aid);
164  strcat((*url), "/");
165  strcat((*url), did);
166  strcat((*url), "/");
167  strcat((*url), sid);
168 
169  free(aid);
170  free(did);
171  free(sid);
172 
173  return HPD_E_SUCCESS;
174 
175  id_error:
176  hpd_adapter_id_free(adapter);
177  hpd_device_id_free(device);
178  return rc;
179 
180  encode_error:
181  free(aid);
182  free(did);
183  free(sid);
184  return rc;
185 }

188 static hpd_error_t url_extract(hpd_rest_t *rest, const char *url, char **aid, char **did, char **sid)
189 {
190  const char *aid_p, *did_p, *sid_p;
191  size_t aid_l, did_l, sid_l;
192 
193  if (strlen(url) < 2)
194  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
195  if (url[0] != '/')
196  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
197 
198  aid_p = &url[1];
199  if (aid_p[0] == '\0' || aid_p[0] == '/')
200  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
201  for (aid_l = 0; aid_p[aid_l] != '/'; aid_l ++)
202  if (aid_p[aid_l] == '\0')
203  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
204 
205  did_p = &aid_p[aid_l+1];
206  if (did_p[0] == '\0' || did_p[0] == '/')
207  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
208  for (did_l = 0; did_p[did_l] != '/'; did_l ++)
209  if (did_p[did_l] == '\0')
210  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
211 
212  sid_p = &did_p[did_l+1];
213  if (sid_p[0] == '\0' || sid_p[0] == '/')
214  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
215  for (sid_l = 0; sid_p[sid_l] != '/' && sid_p[sid_l] != '\0'; sid_l++);
216 
217  if (!sid_p[sid_l] == '\0')
218  HPD_LOG_RETURN(rest->context, HPD_E_ARGUMENT, "URL Parse error.");
219 
220  *aid = NULL;
221  *did = NULL;
222  *sid = NULL;
223  HPD_STR_N_CPY(*aid, aid_p, aid_l);
224  HPD_STR_N_CPY(*did, did_p, did_l);
225  HPD_STR_N_CPY(*sid, sid_p, sid_l);
226 
227  return HPD_E_SUCCESS;
228 
229  alloc_error:
230  free(*aid);
231  free(*did);
232  free(*sid);
233  return HPD_E_ALLOC;
234 }

236 static hpd_error_t reply(hpd_httpd_request_t *req, enum hpd_status status, hpd_rest_req_t *rest_req,
237  hpd_module_t *context)
238 {
239  hpd_error_t rc;
240 
241  if (rest_req && rest_req->http_res)
242  HPD_LOG_RETURN(rest_req->rest->context, HPD_E_STATE, "Already replied to request.");
243 
244  // Check status code
245  switch (status) {
246 #define XX(NUM, STR) case HPD_S_##NUM: break;
247  HPD_HTTP_STATUS_CODE_MAP(XX)
248 #undef XX
249  default:
250  if (context) HPD_LOG_WARN(context, "Unknown HTTP Status code.");
251  status = HPD_S_500;
252  }
253 
254  hpd_httpd_response_t *res;
255  if ((rc = hpd_httpd_response_create(&res, req, status))) return rc;
256 
257  switch (status) {
258 #define XX(NUM, STR) case HPD_S_##NUM: rc = hpd_httpd_response_sendf(res, #STR); break;
259  HPD_HTTP_STATUS_CODE_MAP(XX)
260 #undef XX
261  }
262 
263  if (rc) {
264  hpd_httpd_response_destroy(res);
265  return rc;
266  }
267 
268  if (rest_req) rest_req->http_res = res;
269  return hpd_httpd_response_destroy(res);
270 }

297 static hpd_error_t reply_devices(hpd_rest_req_t *rest_req)
298 {
299  hpd_error_t rc, rc2;
300  hpd_httpd_request_t *http_req = rest_req->http_req;
301  hpd_rest_t *rest = rest_req->rest;
302  hpd_module_t *context = rest->context;
303 
304  if (rest_req->http_res) HPD_LOG_RETURN(context, HPD_E_STATE, "Response already sent.");
305 
306  // Get Accept header
307  const char *accept;
308  if ((rc = hpd_map_get(rest_req->headers, "Accept", &accept))) return rc;
309 
310  // Create body
311  char *body;
312  switch (media_type_to_enum(accept)) {
313  case CONTENT_NONE:
314  case CONTENT_XML:
315  case CONTENT_WILDCARD:
316  if ((rc = hpd_rest_xml_get_configuration(rest->hpd, rest, context, &body))) return rc;
317  break;
318  case CONTENT_JSON:
319  if ((rc = hpd_rest_json_get_configuration(rest->hpd, rest, context, &body))) return rc;
320  break;
321  case CONTENT_UNKNOWN:
322  if ((rc = reply_unsupported_media_type(http_req, rest_req, context))) {
323  HPD_LOG_ERROR(context, "Failed to send unsupported media type response (code: %d).", rc);
324  }
325  return HPD_E_SUCCESS;
326  }
327 
328  // Send response
329  if ((rc = hpd_httpd_response_create(&rest_req->http_res, http_req, HPD_S_200))) goto create_error;
330  if ((rc = hpd_httpd_response_sendf(rest_req->http_res, "%s", body))) goto response_error;
331  if ((rc = hpd_httpd_response_destroy(rest_req->http_res))) goto create_error;
332 
333  free(body);
334 
335  return rc;
336 
337  response_error:
338  if ((rc2 = hpd_httpd_response_destroy(rest_req->http_res)))
339  HPD_LOG_ERROR(context, "Failed to destroy response (code: %d).", rc2);
340  rest_req->http_res = NULL;
341  create_error:
342  free(body);
343  return rc;
344 }

412  return HPD_E_SUCCESS;
413

416  HPD_LOG_ERROR(context, "Failed to destroy response (code: %d).", rc2);

463 static hpd_error_t on_response(hpd_response_t *res)
464 {
465  hpd_error_t rc, rc2;
466 
467  // Get rest request
468  hpd_rest_req_t *rest_req;
469  if ((rc = hpd_response_get_request_data(res, (void **) &rest_req))) return rc;
470 
471  // Make life easier
472  hpd_httpd_request_t *http_req = rest_req->http_req;
473  hpd_module_t *context = rest_req->rest->context;
474 
475  // Check if closed from downstairs
476  if (!http_req) return HPD_E_SUCCESS;
477 
478  // Get data from hpd
479  hpd_value_t *value;
480  const char *val;
481  size_t len;
482  hpd_status_t status;
483  if ((rc = hpd_response_get_value(res, &value)) ||
484  (rc = hpd_value_get_body(value, &val, &len)) ||
485  (rc = hpd_response_get_status(res, &status))) {
486  if ((rc2 = reply_internal_server_error(http_req, rest_req, context))) {
487  HPD_LOG_ERROR(context, "Failed to send internal server error response (code: %d).", rc2);
488  }
489  return rc;
490  }
491 
492  // Check val from hpd
493  if (!val) {
494  if ((rc2 = reply(http_req, status, rest_req, context))) {
495  HPD_LOG_ERROR(context, "Failed to send status response (code: %d).", rc2);
496  }
497  return HPD_E_ARGUMENT;
498  }
499 
500  // Get data from httpd
501  const char *accept;
502  content_type_t accept_type;
503  switch ((rc = hpd_map_get(rest_req->headers, "Accept", &accept))) {
504  case HPD_E_SUCCESS:
505  break;
506  case HPD_E_NOT_FOUND:
507  accept = NULL;
508  break;
509  default:
510  if ((rc2 = reply_internal_server_error(http_req, rest_req, context))) {
511  HPD_LOG_ERROR(context, "Failed to send internal server error response (code: %d).", rc2);
512  }
513  return rc;
514  }
515  switch ((accept_type = media_type_to_enum(accept))) {
516  case CONTENT_NONE:
517  case CONTENT_XML:
518  case CONTENT_JSON:
519  case CONTENT_WILDCARD:
520  break;
521  case CONTENT_UNKNOWN:
522  if ((rc2 = reply_unsupported_media_type(http_req, rest_req, context))) {
523  HPD_LOG_ERROR(context, "Failed to send unsupport1ed media type response (code: %d).", rc2);
524  }
525  return HPD_E_SUCCESS;
526  }
527 
528  // Null terminate val
529  char *body = NULL;
530  body = malloc((len+1) * sizeof(char));
531  if (!body) {
532  if ((rc2 = reply_internal_server_error(http_req, rest_req, context))) {
533  HPD_LOG_ERROR(context, "Failed to send internal server error response (code: %d).", rc2);
534  }
535  HPD_LOG_RETURN_E_ALLOC(context);
536  }
537  strncpy(body, val, len);
538  body[len] = '\0';
539 
540  if (rest_req->http_res) {
541  HPD_LOG_ERROR(context, "Response already sent.");
542  rc = HPD_E_STATE;
543  goto error_free_body;
544  }
545 
546  // Create response
547  if ((rc = hpd_httpd_response_create(&rest_req->http_res, http_req, status))) goto error_send_res;
548  hpd_httpd_response_t *http_res = rest_req->http_res;
549  char *state = NULL;
550  switch (accept_type) {
551  case CONTENT_NONE:
552  case CONTENT_XML:
553  case CONTENT_WILDCARD:
554  if ((rc = hpd_rest_xml_get_value(body, context, &state))) goto error_free_res;
555  if ((rc = hpd_httpd_response_add_header(http_res, "Content-Type", "application/xml"))) goto error_free_state;
556  break;

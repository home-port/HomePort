205 #define FIND_ADAPTER(ID, ADAPTER) do { \
206  (ADAPTER) = NULL; \
207  HPD_TAILQ_FOREACH((ADAPTER), &(ID)->hpd->configuration->adapters) \
208  if (strcmp((ADAPTER)->id, (ID)->aid) == 0) break; \
209  if (!(ADAPTER)) return HPD_E_NOT_FOUND; \
210 } while (0)

213  hpd_adapter_t *a = NULL; \
214  FIND_ADAPTER(&(ID)->adapter, a); \
215  (DEVICE) = NULL; \
216  HPD_TAILQ_FOREACH((DEVICE), a->devices) \
217  if (strcmp((DEVICE)->id, (ID)->did) == 0) break; \
218  if (!(DEVICE)) return HPD_E_NOT_FOUND; \
219 } while (0)

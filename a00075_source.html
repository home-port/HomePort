36 hpd_error_t value_alloc(hpd_value_t **value, const char *body, int len)
37 {
38  hpd_error_t rc;
39  HPD_CALLOC((*value), 1, hpd_value_t);
40  if ((rc = hpd_map_alloc(&(*value)->headers))) {
41  free(*value);
42  return rc;
43  }
44  if (body) {
45  HPD_STR_CPY((*value)->body, body);
46  if (len == HPD_NULL_TERMINATED) (*value)->len = strlen(body);
47  else (*value)->len = (size_t) len;
48  }
49  return HPD_E_SUCCESS;
50 
51  alloc_error:
52  value_free(*value);
53  (*value) = NULL;
54  LOG_RETURN_E_ALLOC();
55 }

57 hpd_error_t value_copy(hpd_value_t **dst, const hpd_value_t *src)
58 {
59  hpd_error_t rc;
60  if ((rc = value_alloc(dst, src->body, (int) src->len))) return rc;
61  hpd_pair_t *pair;
62  hpd_map_foreach(rc, pair, src->headers) {
63  const char *k, *v;
64  if ((rc = hpd_pair_get(pair, &k, &v))) {
65  value_free(*dst);
66  return rc;
67  }
68  if ((rc = value_set_header(*dst, k, v))) {
69  value_free(*dst);
70  return rc;
71  }
72  }
73  return rc;
74 }
